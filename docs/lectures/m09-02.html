Hierarchical clustering

set.seed(1)
penguins_ = penguins %>% mutate_if(is.numeric, (function(x) scale(x, center=F)))

hc_spec <- hier_clust(
  num_clusters = 3,
  linkage_method = "average"
)

hc_fit <- hc_spec %>% 
  fit(~ bill_length_mm + bill_depth_mm + flipper_length_mm + body_mass_g,
    data = penguins_)

hc_fit$fit %>% plot()

Tuning k

penguins_cv <- vfold_cv(penguins_, v = 5)

hc_spec <- hier_clust(
  num_clusters = tune(),
  linkage_method = "average"
)

clust_num_grid <- grid_regular(
  num_clusters(),
  levels = 10
)

penguins_rec <- recipe(~ bill_length_mm + bill_depth_mm + flipper_length_mm + body_mass_g,
  data = penguins_
)

hc_wf <- workflow() %>% 
  add_recipe(penguins_rec) %>% 
  add_model(hc_spec)

res <- tune_cluster(
  hc_wf,
  resamples = penguins_cv,
  grid = clust_num_grid,
  metrics = cluster_metric_set(sse_ratio, silhouette_avg)
)

k-means clustering

kmeans_spec <- k_means(
  num_clusters = 3
)

kmeans_fit <- kmeans_spec %>% 
  fit(~ bill_length_mm + bill_depth_mm + flipper_length_mm + body_mass_g,
    data = penguins_)

km_clusters <- cbind(penguins_, extract_cluster_assignment(kmeans_fit))

Clustering on PCA data

pca_rec <- recipe(~ bill_length_mm + bill_depth_mm + flipper_length_mm + body_mass_g, penguins_) %>% 
  step_pca(all_predictors())

pca_estimates <- prep(pca_rec, penguins_)

pca_data <- cbind(penguins_, bake(pca_estimates, penguins_))

kmeans_spec <- k_means(
  num_clusters = 3
)

kmeans_fit <- kmeans_spec %>% 
  fit(~ PC1 + PC2 + PC3 + PC4,
  data = pca_data)

km_clusters <- cbind(pca_data, extract_cluster_assignment(kmeans_fit))

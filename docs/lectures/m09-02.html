Tuning k

penguins_cv <- vfold_cv(penguins, v = 5)

hc_tune_spec <- hier_clust(
  num_clusters = tune(),
  linkage_method = "average"
)

clust_num_grid <- grid_regular(
  num_clusters(),
  levels = 10
)

hc_tune_wf <- workflow() %>% 
  add_recipe(rec) %>% 
  add_model(hc_tune_spec)

res <- tune_cluster(
  hc_tune_wf,
  resamples = penguins_cv,
  grid = clust_num_grid,
  metrics = cluster_metric_set(sse_ratio, silhouette_avg)
)
